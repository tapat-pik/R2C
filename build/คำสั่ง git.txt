# 1. check status
git status

# 2. เริ่มต้น git (ถ้ายังไม่ได้ทำ)
git init

# 3. เพิ่ม Remote ชื่อ origin ด้วย URL นี้
git remote add origin https://github.com/tapat-pik/R2C.git

# 4. ดึงไฟล์ทั้งหมดเข้าเครื่อง
git pull origin main

# 5. push ไฟล์ที่แก้ไขขึ้น git
git add .
git commit -m "update filter upcoming"
git push origin main
